import java.io.*;
import java.util.*;
import java.text.*;
import java.math.*;
import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        /* Enter your code here. Read input from STDIN. Print output to STDOUT. Your class should be named Solution. */
        
        Scanner in = new Scanner(System.in);
        int t = in.nextInt();
        for(int a0 = 0; a0 < t; a0++){
            int A = in.nextInt();
            int B = in.nextInt();
            int countSqInt = 0;
            double potentialSq = 1;
            
            for (int i = A; i <= B; i++){
                
                   
                    potentialSq = (potentialSq + ((Math.sqrt(potentialSq))* 2) ) + 1;
                
                    if ( potentialSq % (Math.sqrt(potentialSq)) ==0 ) {
                        
                     
                             countSqInt++;
                        
                             i =  (int) potentialSq;
                    }
                
            }
            
            System.out.println(countSqInt);
            potentialSq = 1;
            countSqInt = 0;
        }
    }
}
